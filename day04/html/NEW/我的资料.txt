	$(".file_tab").each(function(i) {
        $(this).click(function (){

				$(".tab_li").css("max-height","0");
				$(".tab_li").eq(i).css("max-height","945px");
				$(".file_tab i").css("transform","rotate(0deg)");
				$(".file_tab i").eq(i).css("transform","rotate(180deg)");
			} else{
				$(".tab_li").eq(i).css("max-height","0");
				$(".file_tab i").eq(i).css("transform","rotate(0deg)");
			}
			if( i!==6 ){
				$(".xxzl").css("height","0");
				$(".xxzl_tt i").css("transform","rotate(0deg)");
			}
		})

    });
	$(".xxzl_tt").click(function (){
		if( $(".xxzl").height()<=0 ){
			$(".file_up").css("max-height","1850px");

			$(".xxzl_tt i").css("transform","rotate(180deg)");
		} else{
			$(".file_up").css("max-height","945px");
			$(".xxzl").css("height","0");
			$(".xxzl_tt i").css("transform","rotate(0deg)");
		}
	})
})
